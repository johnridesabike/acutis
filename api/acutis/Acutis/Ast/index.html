The echo syntax is essentially a subset of the pattern syntax, one which only allows strings, variables, and record field access.

type echo =
  1. | Echo_var of Loc.t * string
  2. | Echo_string of Loc.t * string
  3. | Echo_field of echo * string
type pat =
  1. | Var of Loc.t * string
  2. | Bool of Loc.t * int
  3. | Int of Loc.t * int
  4. | Float of Loc.t * float
  5. | String of Loc.t * string
  6. | Nullable of Loc.t * pat option
  7. | Enum_string of Loc.t * string
  8. | Enum_int of Loc.t * int
  9. | List of Loc.t * pat list * pat option
  10. | Tuple of Loc.t * pat list
  11. | Record of Loc.t * pat record
  12. | Dict of Loc.t * pat assoc
  13. | Block of Loc.t * t
  14. | Field of Loc.t * pat * string
and node =
  1. | Text of string * trim * trim
  2. | Echo of (echo_format * echo) list * echo_format * echo * escape
  3. | Match of Loc.t * pat Nonempty.t * case Nonempty.t
  4. | Map_list of Loc.t * pat * case Nonempty.t
  5. | Map_dict of Loc.t * pat * case Nonempty.t
  6. | Component of Loc.t * string * string * pat assoc
  7. | Interface of Loc.t * interface
  8. | Comment of string
and case = {
  1. pats : (Loc.t * pat Nonempty.t) Nonempty.t;
  2. nodes : t;
}
and t = node list
