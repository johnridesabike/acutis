Module Acutis.Typechecker

Type-checks the untyped Ast.t and constructs a typed tree.

type echo =
  1. | Echo_var of string
  2. | Echo_string of string
  3. | Echo_field of echo * string
type construct =
  1. | TList
  2. | TNullable
type pat =
  1. | TConst of Data.Const.t * Typescheme.Enum.t option
  2. | TConstruct of construct * pat option
  3. | TTuple of pat list
  4. | TRecord of (string * Data.Const.t * Typescheme.t Typescheme.Union.t) option * pat Map.String.t * Typescheme.t Map.String.t Stdlib.ref
  5. | TDict of pat Map.String.t * Set.String.t Stdlib.ref
  6. | TVar of string
  7. | TBlock of nodes
  8. | TField of pat * string
  9. | TAny
and node =
  1. | TText of string * Ast.trim * Ast.trim
  2. | TEcho of (Ast.echo_format * echo) list * Ast.echo_format * echo * Ast.escape
  3. | TMatch of Loc.t * pat Nonempty.t * Typescheme.t Nonempty.t * case Nonempty.t
  4. | TMap_list of Loc.t * pat * Typescheme.t Nonempty.t * case Nonempty.t
  5. | TMap_dict of Loc.t * pat * Typescheme.t Nonempty.t * case Nonempty.t
  6. | TComponent of string * pat Map.String.t
and case = {
  1. pats : (Loc.t * pat Nonempty.t) Nonempty.t;
  2. nodes : nodes;
}
and nodes = node list
type t = {
  1. nodes : nodes;
  2. types : Typescheme.t Map.String.t;
}
type ('a, 'b) source =
  1. | Src of string * 'a
  2. | Fun of string * Typescheme.t Map.String.t * 'b
val make_components : (Ast.t, 'a) source Map.String.t -> (t, 'a) source Map.String.t
val make : root:string -> (t, 'a) source Map.String.t -> Ast.t -> t
val make_interface_standalone : Ast.interface -> Typescheme.t Map.String.t
